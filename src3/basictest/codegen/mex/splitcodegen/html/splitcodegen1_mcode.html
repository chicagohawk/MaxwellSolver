1% all combinations that put n numbers into m bits
 2%   e.g. n = 3, m = 3, returns
 3%     003 300 030 102 012 201 111 021 210 120
 4function bits = splitcodegen(n, m)
 5    if n == 0,
 6        bits = zeros(1,m);
 7        return;
 8    end
 9    
10    if m == 1,
11        bits = n;
12        return;
13    end
14    
15    bits = [];
16    for ii = 0 : n,    % number put in first bit
17        subcode = codegen(n-ii, m-1);
18        num = size(subcode,1);
19        combcode = [ii * ones(num,1), subcode];
20        bits = [bits; combcode];
21    end
22    
23    return;
24end
